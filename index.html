import tkinter as tk
from tkinter import messagebox

class TodoApp:
    def __init__(self, ventana):
        self.ventana = ventana
        self.ventana.title("游닇 Lista de Tareas")
        self.ventana.geometry("400x500")
        self.ventana.config(bg="#f0f0f0")
        
        # Frame superior para entrada
        frame_entrada = tk.Frame(self.ventana, bg="#f0f0f0")
        frame_entrada.pack(pady=10)

        self.entrada.pack(side="left", padx=5)
        self.entrada.bind("<Return>", lambda e: self.agregar_tarea())
        
        btn_agregar = tk.Button(frame_entrada, text="俱 Agregar",
                                command=self.agregar_tarea,
                                bg="#4CAF50", fg="white")
        btn_agregar.pack(side="left")
        
        # Listbox para tareas
        self.listbox = tk.Listbox(self.ventana, 
                                  width=40, height=15,
                                  font=("Arial", 11))
        self.listbox.pack(pady=10)
        
        # Frame para botones de acci칩n
        frame_botones = tk.Frame(self.ventana, bg="#f0f0f0")
        frame_botones.pack()
        
        tk.Button(frame_botones, text="仇 Eliminar",
                  command=self.eliminar_tarea,
                  bg="#f44336", fg="white", width=12).pack(side="left", padx=5)
        
        tk.Button(frame_botones, text="游딈勇 Limpiar Todo",
                  command=self.limpiar_todo,
                  bg="#FF9800", fg="white", width=12).pack(side="left")
        
    def agregar_tarea(self):
        tarea = self.entrada.get()
        if tarea:
            self.listbox.insert(tk.END, tarea)
            self.entrada.delete(0, tk.END)
        else:
            messagebox.showwarning("Advertencia", 
                                   "Escribe una tarea")
    
    def eliminar_tarea(self):
        try:
            indice = self.listbox.curselection()[0]
            self.listbox.delete(indice)
        except:
            messagebox.showwarning("Advertencia", 
                                   "Selecciona una tarea")
    
    def limpiar_todo(self):
        self.listbox.delete(0, tk.END)

Proyecto 2: Calculadora Simple

import tkinter as tk

class Calculadora:
    def __init__(self, ventana):
        self.ventana = ventana
        self.ventana.title("游댝 Calculadora")
        self.ventana.geometry("300x400")
        self.ventana.resizable(False, False)
        
        self.expresion = ""
        
        # Display
        self.texto_var = tk.StringVar()
        display = tk.Entry(ventana, textvariable=self.texto_var,
                         font=("Arial", 20), bd=10,
                         justify="right", state="readonly")
        display.grid(row=0, column=0, columnspan=4, sticky="nsew")
        
        # Botones
        botones = [
            ['7', '8', '9', '/'],
            ['4', '5', '6', '*'],
            ['1', '2', '3', '-'],
            ['C', '0', '=', '+']
        ]
        
        for i, fila in enumerate(botones, start=1):
            for j, boton in enumerate(fila):
                comando = lambda x=boton: self.click(x)
                tk.Button(ventana, text=boton, font=("Arial", 18),
                          command=comando, bg="#e0e0e0").grid(
                              row=i, column=j, sticky="nsew", padx=2, pady=2)
        
        # Configurar pesos de filas/columnas para expansi칩n
        for i in range(5):
            ventana.grid_rowconfigure(i, weight=1)
        for i in range(4):
            ventana.grid_columnconfigure(i, weight=1)
    
    def click(self, boton):
        if boton == 'C':
            self.expresion = ""
        elif boton == '=':
            try:
                resultado = str(eval(self.expresion))
                self.expresion = resultado
            except:
                self.expresion = "Error"
        else:
            self.expresion += boton
        
        self.texto_var.set(self.expresion)

ventana = tk.Tk()
app = Calculadora(ventana)
ventana.mainloop()
